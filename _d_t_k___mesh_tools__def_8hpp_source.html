291 template<class Mesh>
292 BoundingBox MeshTools<Mesh>::localBoundingBox( const Mesh& mesh )
293 {
294  double x_min = -Teuchos::ScalarTraits<double>::rmax();
295  double y_min = -Teuchos::ScalarTraits<double>::rmax();
296  double z_min = -Teuchos::ScalarTraits<double>::rmax();
297 
298  double x_max = Teuchos::ScalarTraits<double>::rmax();
299  double y_max = Teuchos::ScalarTraits<double>::rmax();
300  double z_max = Teuchos::ScalarTraits<double>::rmax();
301 
302  GlobalOrdinal num_nodes = std::distance( MT::nodesBegin( mesh ),
303  MT::nodesEnd( mesh ) );
304  std::size_t node_dim = MT::nodeDim( mesh );
305 
306  if ( node_dim > 0 )
307  {
308  x_min = *std::min_element(
309  MT::coordsBegin( mesh ),
310  MT::coordsBegin( mesh ) + num_nodes );
311  x_max = *std::max_element(
312  MT::coordsBegin( mesh ),
313  MT::coordsBegin( mesh ) + num_nodes );
314  }
315  if ( node_dim > 1 )
316  {
317  y_min = *std::min_element(
318  MT::coordsBegin( mesh ) + num_nodes,
319  MT::coordsBegin( mesh ) + 2*num_nodes );
320  y_max = *std::max_element(
321  MT::coordsBegin( mesh ) + num_nodes,
322  MT::coordsBegin( mesh ) + 2*num_nodes );
323  }
324  if ( node_dim > 2 )
325  {
326  z_min = *std::min_element(
327  MT::coordsBegin( mesh ) + 2*num_nodes,
328  MT::coordsBegin( mesh ) + 3*num_nodes );
329  z_max = *std::max_element(
330  MT::coordsBegin( mesh ) + 2*num_nodes,
331  MT::coordsBegin( mesh ) + 3*num_nodes );
332  }
333 
334  return BoundingBox( x_min, y_min, z_min, x_max, y_max, z_max );
335 }

342 template<class Mesh>
343 BoundingBox MeshTools<Mesh>::globalBoundingBox( const Mesh& mesh,
344  const RCP_Comm& comm )
345 {
346  BoundingBox local_box = localBoundingBox( mesh );
347  Teuchos::Tuple<double,6> local_bounds = local_box.getBounds();
348 
349  double global_x_min, global_y_min, global_z_min;

352  Teuchos::reduceAll<int,double>( *comm,
