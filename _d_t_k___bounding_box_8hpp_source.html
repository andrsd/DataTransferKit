00125 BoundingBox& BoundingBox::operator +=( const BoundingBox& box )
00126 {
00127     // When we do a global reduction with a bounding box array, we only want
00128     // one bounding box in each array segment. They will be unique, but this
00129     // way we will be sure this is true. If the default constructor was called
00130     // on this box, then set this box's bounds to the other box. Otherwise
00131     // we'll return this box. 
00132     if ( d_x_min == 0.0 && d_y_min == 0.0 && d_z_min == 0.0 &&
00133          d_x_max == 0.0 && d_y_max == 0.0 && d_z_max == 0.0 )
00134     {
00135         d_x_min = box.d_x_min;
00136         d_y_min = box.d_y_min;
00137         d_z_min = box.d_z_min;
00138         d_x_max = box.d_x_max;
00139         d_y_max = box.d_y_max;
00140         d_z_max = box.d_z_max;
00141     }
00142 
00143     return *this;
00144 }
