00090 template<typename GlobalOrdinal>
00091 GlobalOrdinal KDTree<GlobalOrdinal>::findPoint( 
00092     const Teuchos::Array<double>& coords )
00093 {
00094     std::size_t node_dim = coords.size();
00095     double point[3];
00096     for ( std::size_t d = 0; d < node_dim; ++d )
00097     {
00098         point[d] = coords[d];
00099     }
00100     for ( std::size_t d = node_dim; d < 3; ++d )
00101     {
00102         point[d] = 0;
00103     }
00104 
00105     moab::ErrorCode error;
00106     moab::EntityHandle leaf;
00107     error = d_tree.leaf_containing_point( d_root, point, leaf );
00108     testInvariant( moab::MB_SUCCESS == error,
00109                    "Failed to search kD-tree." );
00110 
00111     // Try to find the point in the leaf. If we don't then return the maximum
00112     // global ordinal for this type. This assumes that this number will not be
00113     // reached in global number of elements.
00114     try
00115     {
00116         moab::EntityHandle element = findPointInLeaf( coords, leaf );
00117         return d_mesh->getNativeOrdinal( element );
00118     }
00119     catch ( PointNotFound& exception )
00120     {
00121         return Teuchos::OrdinalTraits<GlobalOrdinal>::max();
00122     }
00123 }

00130 template<typename GlobalOrdinal>
00131 moab::EntityHandle KDTree<GlobalOrdinal>::findPointInLeaf( 
00132     const Teuchos::Array<double>& coords, const moab::EntityHandle leaf )
00133 {
00134     moab::ErrorCode error;
00135 
00136     // Get the largest dimension of the elements in the leaf.
00137     int dim = 0;
00138     int num_elements_by_dim = 0;
00139     for ( int d = 0; d < 4; ++d )
00140     {
00141         error = d_mesh->getMoab()->get_number_entities_by_dimension(
00142             leaf, d, num_elements_by_dim );
00143         testInvariant( moab::MB_SUCCESS == error,
00144                        "Failed to get number of elements by dimension." );
00145 
00146         if ( num_elements_by_dim > 0 )
00147         {
00148             dim = d;
00149         }
00150     }
00151 
00152     // Get the elements in the leaf.
00153     std::vector<moab::EntityHandle> leaf_elements;
00154     error = d_mesh->getMoab()->get_entities_by_dimension( 
00155         leaf, dim, leaf_elements );
00156     testInvariant( moab::MB_SUCCESS == error,
00157                    "Failed to get leaf elements" );
00158 
00159     // Search the leaf elements with the point.
00160     Teuchos::Array<double> point( coords );
00161     std::vector<moab::EntityHandle>::const_iterator leaf_iterator;
00162     for ( leaf_iterator = leaf_elements.begin();
00163           leaf_iterator != leaf_elements.end();
00164           ++leaf_iterator )
00165     {
00166         if ( TopologyTools::pointInElement( 
00167                  point, *leaf_iterator, d_mesh->getMoab() ) )
00168         {
00169             return *leaf_iterator;
00170         }
00171     }
00172 
00173     // We didn't find the point so we'll throw an exception.
00174     throw PointNotFound( "Point not found in kD-tree." );
00175 
00176     // Return a 0 handle if no element was found.
00177     return 0;
00178 }
