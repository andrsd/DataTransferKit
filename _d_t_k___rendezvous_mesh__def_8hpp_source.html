86 template<class Mesh>
87 Teuchos::RCP< RendezvousMesh<typename MeshTraits<Mesh>::global_ordinal_type> >
88 createRendezvousMesh( const MeshManager<Mesh>& mesh_manager )
89 {
90  // Setup types and iterators as we're outside of the class definition.
91  typedef MeshTraits<Mesh> MT;
92  typedef typename MT::global_ordinal_type GlobalOrdinal;
93  typename MT::const_node_iterator node_iterator;
94  typename MT::const_element_iterator element_iterator;
95 
96  // Setup a moab range for the mesh elements.
97  moab::Range moab_elements;
98 
99  // Setup an element handle map.
100  std::map<moab::EntityHandle,GlobalOrdinal> element_handle_map;
101 
102  // Create a moab interface.
103  moab::ErrorCode error;
104  Teuchos::RCP<moab::Interface> moab = Teuchos::rcp( new moab::Core() );
105  testPostcondition( moab != Teuchos::null,
106  "Error creating MOAB interface" );
107 
108  // Set the mesh dimension.
109  std::size_t node_dim = mesh_manager.dim();
110  error = moab->set_dimension( node_dim );
111 
112  // Build each mesh block.
113  typename MeshManager<Mesh>::BlockIterator block_iterator;
114  for ( block_iterator = mesh_manager.blocksBegin();
115  block_iterator != mesh_manager.blocksEnd();
116  ++block_iterator )
117  {
118  // Check the nodes and coordinates for consistency.
119  GlobalOrdinal num_nodes = MeshTools<Mesh>::numNodes( *block_iterator );
120  GlobalOrdinal num_coords =
121  std::distance( MT::coordsBegin( *block_iterator ),
122  MT::coordsEnd( *block_iterator ) );
123  testInvariant(
124  num_coords == (GlobalOrdinal) node_dim * num_nodes,
125  "Number of coordinates provided != node_dim * number of nodes" );
126 
127  // Add the mesh nodes to moab and map the native vertex handles to the
128  // moab vertex handles. This should be in a hash table. We'll need one
129  // that hashes moab handles.
130  double vertex_coords[3];
131  Teuchos::ArrayRCP<const double> mesh_coords =
132  MeshTools<Mesh>::coordsView( *block_iterator );
133  std::map<GlobalOrdinal,moab::EntityHandle> vertex_handle_map;
134  GlobalOrdinal n = 0;
135  for ( node_iterator = MT::nodesBegin( *block_iterator );
136  node_iterator != MT::nodesEnd( *block_iterator );
137  ++node_iterator, ++n )
138  {
139  moab::EntityHandle moab_vertex;
140  for ( std::size_t d = 0; d < node_dim; ++d )
141  {
142  vertex_coords[d] = mesh_coords[d*num_nodes + n];
143  }
144  for ( std::size_t d = node_dim; d < 3; ++d )
145  {
146  vertex_coords[d] = 0.0;
147  }
148  error = moab->create_vertex( vertex_coords, moab_vertex );
149  testInvariant( moab::MB_SUCCESS == error,
150  "Failed to create vertices in MOAB." );
151  vertex_handle_map[ *node_iterator ] = moab_vertex;
152  }
153 
154  // Check the elements and connectivity for consistency.
155  int nodes_per_element =
156  MT::nodesPerElement( *block_iterator );
157  GlobalOrdinal num_elements =
158  MeshTools<Mesh>::numElements( *block_iterator );
159  GlobalOrdinal num_connect =
160  std::distance( MT::connectivityBegin( *block_iterator ),
161  MT::connectivityEnd( *block_iterator ) );
162  testPrecondition(
163  num_elements == num_connect / nodes_per_element &&
164  num_connect % nodes_per_element == 0,
165  "Connectivity array inconsistent with element description." );
166 
167  // Extract the mesh elements and add them to moab.
168  Teuchos::ArrayRCP<const GlobalOrdinal> mesh_connectivity =
169  MeshTools<Mesh>::connectivityView( *block_iterator );
170  Teuchos::ArrayRCP<const std::size_t> permutation_list =
171  MeshTools<Mesh>::permutationView( *block_iterator );
172  GlobalOrdinal conn_index;
173  Teuchos::Array<moab::EntityHandle> element_connectivity( nodes_per_element );
174 
175  std::size_t canonical_idx;
176  n = 0;
177  int element_topology = MT::elementTopology( *block_iterator );
178  for ( element_iterator = MT::elementsBegin( *block_iterator );
179  element_iterator != MT::elementsEnd( *block_iterator );
180  ++element_iterator, ++n )
181  {
182  // Extract the connecting nodes for this element and apply the
183  // permutation list.
184  for ( int i = 0; i < nodes_per_element; ++i )
185  {
186  canonical_idx = permutation_list[i];
187  conn_index = i*num_elements + n;
188  element_connectivity[ canonical_idx ] =
189  vertex_handle_map.find(
190  mesh_connectivity[ conn_index ] )->second;
191  }
192  testInvariant( (int) element_connectivity.size() == nodes_per_element,
193  "Element connectivity size != nodes per element." );
194 
195  // Create the element in moab.
196  moab::EntityType entity_type =
197  moab_topology_table[ element_topology ];
198  moab::EntityHandle moab_element;
199  error = moab->create_element( entity_type,
200  &element_connectivity[0],
201  element_connectivity.size(),
202  moab_element );
203  testInvariant( moab::MB_SUCCESS == error,
204  "Failed to create element in MOAB." );
205  moab_elements.insert( moab_element );
206 
207  // Map the moab element handle to the native element handle.
208  element_handle_map[ moab_element ] = *element_iterator;
209  }
210  }
211 
212  // Create and return the mesh.
213  return Teuchos::rcp( new RendezvousMesh<GlobalOrdinal>(
214  moab, moab_elements, element_handle_map ) );
215 }
