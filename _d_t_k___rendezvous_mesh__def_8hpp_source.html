00093 template<class Mesh>
00094 Teuchos::RCP< RendezvousMesh<typename MeshTraits<Mesh>::global_ordinal_type> >
00095 createRendezvousMesh( const MeshManager<Mesh>& mesh_manager )
00096 {
00097     // Setup types and iterators as we're outside of the class definition.
00098     typedef MeshTraits<Mesh> MT;
00099     typedef typename MT::global_ordinal_type GlobalOrdinal;
00100     typename MT::const_vertex_iterator vertex_iterator;
00101     typename MT::const_element_iterator element_iterator;
00102 
00103     // Setup an element handle map.
00104     std::map<moab::EntityHandle,GlobalOrdinal> element_handle_map;
00105 
00106     // Create a moab interface.
00107     moab::ErrorCode error;
00108     Teuchos::RCP<moab::Interface> moab = Teuchos::rcp( new moab::Core() );
00109     testPostcondition( moab != Teuchos::null );
00110 
00111     // Set the mesh dimension.
00112     int vertex_dim = mesh_manager.dim();
00113     error = moab->set_dimension( vertex_dim );
00114     testInvariant( moab::MB_SUCCESS == error );
00115 
00116     // Build each mesh block.
00117     typename MeshManager<Mesh>::BlockIterator block_iterator;
00118     for ( block_iterator = mesh_manager.blocksBegin();
00119           block_iterator != mesh_manager.blocksEnd();
00120           ++block_iterator )
00121     {
00122         // Check the vertices and coordinates for consistency.
00123         GlobalOrdinal num_vertices = 
00124             MeshTools<Mesh>::numVertices( *block_iterator );
00125         GlobalOrdinal num_coords = 
00126             std::distance( MT::coordsBegin( *block_iterator ),
00127                            MT::coordsEnd( *block_iterator ) );
00128         testInvariant( num_coords == 
00129                        (GlobalOrdinal) vertex_dim * num_vertices );
00130 
00131         // Add the mesh vertices to moab and map the native vertex handles to
00132         // the moab vertex handles. This should be in a hash table. We'll need
00133         // one that hashes moab handles.
00134         double vertex_coords[3];
00135         Teuchos::ArrayRCP<const double> mesh_coords = 
00136             MeshTools<Mesh>::coordsView( *block_iterator );
00137         std::map<GlobalOrdinal,moab::EntityHandle> vertex_handle_map;
00138         GlobalOrdinal n = 0;
00139         for ( vertex_iterator = MT::verticesBegin( *block_iterator );
00140               vertex_iterator != MT::verticesEnd( *block_iterator );
00141               ++vertex_iterator, ++n )
00142         {
00143             moab::EntityHandle moab_vertex;
00144             for ( int d = 0; d < vertex_dim; ++d )
00145             {
00146                 vertex_coords[d] = mesh_coords[d*num_vertices + n];
00147             }
00148             for ( int d = vertex_dim; d < 3; ++d )
00149             {
00150                 vertex_coords[d] = 0.0;
00151             }
00152             error = moab->create_vertex( vertex_coords, moab_vertex );
00153             testInvariant( moab::MB_SUCCESS == error );
00154 
00155             vertex_handle_map[ *vertex_iterator ] = moab_vertex;
00156         }
00157 
00158         // Check the elements and connectivity for consistency.
00159         int vertices_per_element = 
00160             MT::verticesPerElement( *block_iterator );
00161         GlobalOrdinal num_elements = 
00162             MeshTools<Mesh>::numElements( *block_iterator );
00163         GlobalOrdinal num_connect = 
00164             std::distance( MT::connectivityBegin( *block_iterator ),
00165                            MT::connectivityEnd( *block_iterator ) );
00166         testInvariant( num_elements == num_connect / vertices_per_element &&
00167                        num_connect % vertices_per_element == 0 );
00168 
00169         // Extract the mesh elements and add them to moab.
00170         Teuchos::ArrayRCP<const GlobalOrdinal> mesh_connectivity = 
00171             MeshTools<Mesh>::connectivityView( *block_iterator );
00172         Teuchos::ArrayRCP<const int> permutation_list =
00173             MeshTools<Mesh>::permutationView( *block_iterator );
00174         GlobalOrdinal conn_index;
00175         Teuchos::Array<moab::EntityHandle> 
00176             element_connectivity( vertices_per_element );
00177 
00178         int canonical_idx;
00179         n = 0;
00180         int element_topology = MT::elementTopology( *block_iterator );
00181         for ( element_iterator = MT::elementsBegin( *block_iterator );
00182               element_iterator != MT::elementsEnd( *block_iterator );
00183               ++element_iterator, ++n )
00184         {
00185             // Extract the connecting vertices for this element and apply the
00186             // permutation list.
00187             for ( int i = 0; i < vertices_per_element; ++i )
00188             {
00189                 canonical_idx = permutation_list[i];
00190                 conn_index = i*num_elements + n;
00191                 element_connectivity[ canonical_idx ] =
00192                     vertex_handle_map.find( 
00193                         mesh_connectivity[ conn_index ] )->second;
00194             }
00195             testInvariant( (int) element_connectivity.size() 
00196                            == vertices_per_element );
00197 
00198             // Create the element in moab.
00199             moab::EntityType entity_type = 
00200                 moab_topology_table[ element_topology ];
00201             moab::EntityHandle moab_element;
00202             error = moab->create_element( entity_type,
00203                                           &element_connectivity[0],
00204                                           element_connectivity.size(),
00205                                           moab_element );
00206             testInvariant( moab::MB_SUCCESS == error );
00207 
00208             // Map the moab element handle to the native element handle.
00209             element_handle_map[ moab_element ] = *element_iterator;
00210         }
00211     }
00212 
00213     // Create and return the mesh.
00214     return Teuchos::rcp( 
00215         new RendezvousMesh<GlobalOrdinal>( moab, element_handle_map ) );
00216 }
