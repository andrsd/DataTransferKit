00317                                 SFT::dim( evaluated_field );

00320                                       GlobalOrdinal> > source_vector = 
00321         createMultiVectorFromView( d_export_map,

00323                                    source_size,
00324                                    SFT::dim( evaluated_field ) );

00356 template<class Mesh, class CoordinateField>
00357 Teuchos::Array<
00358     typename ConsistentEvaluation<Mesh,CoordinateField>::GlobalOrdinal>
00359 ConsistentEvaluation<Mesh,CoordinateField>::computePointOrdinals(
00360     const CoordinateField& coordinate_field )
00361 {
00362     int comm_rank = d_comm->getRank();
00363     int point_dim = CFT::dim( coordinate_field );
00364     GlobalOrdinal local_size = 
00365         std::distance( CFT::begin( coordinate_field ),
00366                        CFT::end( coordinate_field ) ) / point_dim;
00367 
00368     GlobalOrdinal global_size;
00369     Teuchos::reduceAll<int,GlobalOrdinal>( *d_comm,
00370                                            Teuchos::REDUCE_MAX,
00371                                            1,
00372                                            &local_size,
00373                                            &global_size );
00374 
00375     Teuchos::Array<GlobalOrdinal> point_ordinals( local_size );
00376     for ( GlobalOrdinal n = 0; n < local_size; ++n )
00377     {
00378         point_ordinals[n] = comm_rank*global_size + n;
00379     }
00380     return point_ordinals;
00381 }
