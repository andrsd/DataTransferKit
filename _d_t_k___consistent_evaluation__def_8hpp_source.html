84 template<class Mesh, class CoordinateField>
85 void ConsistentEvaluation<Mesh,CoordinateField>::setup(
86  const RCP_MeshManager& mesh_manager,
87  const CoordinateField& coordinate_field )
88 {
89  // Get the global bounding box for the mesh.
90  BoundingBox mesh_box = mesh_manager->globalBoundingBox();
91 
92  // Get the global bounding box for the coordinate field.
93  BoundingBox coord_box = FieldTools<CoordinateField>::coordGlobalBoundingBox(
94  coordinate_field, d_comm );
95 
96  // Intersect the boxes to get the rendezvous bounding box.
97  BoundingBox rendezvous_box;
98  bool has_intersect =
99  BoundingBox::intersectBoxes( mesh_box, coord_box, rendezvous_box );
100  if ( !has_intersect )
101  {
102  throw MeshException(
103  "Mesh and coordinate field domains do not intersect." );
104  }
105 
106  // Build a rendezvous decomposition with the source mesh.
107  Rendezvous<Mesh> rendezvous( d_comm, rendezvous_box );
108  rendezvous.build( mesh_manager );
109 
110  // Compute a unique global ordinal for each point in the coordinate field.
111  Teuchos::Array<GlobalOrdinal> point_ordinals =
112  computePointOrdinals( coordinate_field );
113 
114  // Build the data import map from the point global ordinals.
115  Teuchos::ArrayView<const GlobalOrdinal> import_ordinal_view =
116  point_ordinals();
117  d_import_map = Tpetra::createNonContigMap<GlobalOrdinal>(
118  import_ordinal_view, d_comm );
119  testPostcondition( d_import_map != Teuchos::null,
120  "Error creating data import map." );
121 
122  // Determine the rendezvous destination proc of each point in the
123  // coordinate field.
124  std::size_t coord_dim = CFT::dim( coordinate_field );
125  typename CFT::size_type num_coords = CFT::size( coordinate_field );
126  Teuchos::ArrayRCP<typename CFT::value_type> coords_view;
127  if ( num_coords == 0 )
128  {
129  coords_view = Teuchos::ArrayRCP<typename CFT::value_type>( 0, 0.0 );
130  }
131  else
132  {
133  coords_view =
134  FieldTools<CoordinateField>::nonConstView( coordinate_field );
135  }
136  Teuchos::Array<int> rendezvous_procs =
137  rendezvous.getRendezvousProcs( coords_view );
138 
139  // Via an inverse communication operation, move the global point ordinals
140  // to the rendezvous decomposition.
141  Tpetra::Distributor point_distributor( d_comm );
142  GlobalOrdinal num_rendezvous_points =
143  point_distributor.createFromSends( rendezvous_procs() );
144  Teuchos::Array<GlobalOrdinal>
145  rendezvous_points( num_rendezvous_points );
146  point_distributor.doPostsAndWaits(
147  import_ordinal_view, 1, rendezvous_points() );
148 
149  // Setup Target-to-rendezvous communication.
150  Teuchos::ArrayView<const GlobalOrdinal> rendezvous_points_view =
151  rendezvous_points();
152  RCP_TpetraMap rendezvous_import_map =
153  Tpetra::createNonContigMap<GlobalOrdinal>(
154  rendezvous_points_view, d_comm );
155  Tpetra::Export<GlobalOrdinal>
156  point_exporter( d_import_map, rendezvous_import_map );
157 
158  // Move the target coordinates to the rendezvous decomposition.
159  GlobalOrdinal num_points = num_coords / coord_dim;
160  Teuchos::RCP< Tpetra::MultiVector<double,GlobalOrdinal> >
161  target_coords = createMultiVectorFromView( d_import_map, coords_view,
162  num_points, coord_dim );
163  Tpetra::MultiVector<double,GlobalOrdinal> rendezvous_coords(
164  rendezvous_import_map, coord_dim );
165  rendezvous_coords.doExport( *target_coords, point_exporter,
166  Tpetra::INSERT );
167 
168  // Search the rendezvous decomposition with the target points to get the
169  // source elements that contain them.
170  Teuchos::Array<GlobalOrdinal> rendezvous_elements =
171  rendezvous.elementsContainingPoints(
172  rendezvous_coords.get1dViewNonConst() );
173 
174  // Build a unique list of rendezvous elements.
175  Teuchos::Array<GlobalOrdinal> rendezvous_element_set =
176  rendezvous_elements;
177  std::sort( rendezvous_element_set.begin(), rendezvous_element_set.end() );
178  typename Teuchos::Array<GlobalOrdinal>::iterator unique_bound =
179  std::unique( rendezvous_element_set.begin(),
180  rendezvous_element_set.end() );
181  rendezvous_element_set.resize(
182  unique_bound - rendezvous_element_set.begin() );
183 
184  // Setup source-to-rendezvous communication.
185  Teuchos::ArrayView<const GlobalOrdinal> rendezvous_elem_set_view =
186  rendezvous_element_set();
187  RCP_TpetraMap rendezvous_element_map =
188  Tpetra::createNonContigMap<GlobalOrdinal>(
189  rendezvous_elem_set_view, d_comm );
190 
191  // The block strategy means that we have to make a temporary copy of the
192  // element handles so that we can make a Tpetra::Map.
193  Teuchos::Array<GlobalOrdinal>
194  mesh_elements( mesh_manager->localNumElements() );
195  GlobalOrdinal start = 0;
196  BlockIterator block_iterator;
197  for ( block_iterator = mesh_manager->blocksBegin();
198  block_iterator != mesh_manager->blocksEnd();
199  ++block_iterator )
200  {
201  Teuchos::ArrayRCP<const GlobalOrdinal> mesh_element_arcp =
202  MeshTools<Mesh>::elementsView( *block_iterator );
203  std::copy( mesh_element_arcp.begin(), mesh_element_arcp.end(),
204  mesh_elements.begin() + start );
205  start += mesh_element_arcp.size();
206  }
207 
208  Teuchos::ArrayView<const GlobalOrdinal> mesh_element_view =
209  mesh_elements();
210  RCP_TpetraMap mesh_element_map =
211  Tpetra::createNonContigMap<GlobalOrdinal>(
212  mesh_element_view, d_comm );
213  mesh_elements.clear();
214 
215  Tpetra::Import<GlobalOrdinal> source_importer(
216  mesh_element_map, rendezvous_element_map );
217 
218  // Elements send their source decomposition proc to the rendezvous
219  // decomposition.
220  Tpetra::MultiVector<int,GlobalOrdinal>
221  source_procs( mesh_element_map, 1 );
222  source_procs.putScalar( d_comm->getRank() );
223  Tpetra::MultiVector<int,GlobalOrdinal>
224  rendezvous_source_procs( rendezvous_element_map, 1 );
225  rendezvous_source_procs.doImport( source_procs, source_importer,
226  Tpetra::INSERT );
227 
228  // Set the destination procs for the rendezvous-to-source communication.
229  Teuchos::ArrayRCP<const int> rendezvous_source_procs_view =
230  rendezvous_source_procs.get1dView();
231  GlobalOrdinal num_rendezvous_elements = rendezvous_elements.size();
232  GlobalOrdinal dest_proc_idx;
233  Teuchos::Array<int>
234  target_destinations( num_rendezvous_elements );
235  for ( GlobalOrdinal n = 0; n < num_rendezvous_elements; ++n )
236  {
237  dest_proc_idx = std::distance(
238  rendezvous_element_set.begin(),
239  std::find( rendezvous_element_set.begin(),
240  rendezvous_element_set.end(),
241  rendezvous_elements[n] ) );
242  target_destinations[n] = rendezvous_source_procs_view[ dest_proc_idx ];
243  }
244  rendezvous_element_set.clear();
245 
246  // Send the rendezvous elements to the source decomposition via inverse
247  // communication.
248  Teuchos::ArrayView<const GlobalOrdinal> rendezvous_elements_view =
249  rendezvous_elements();
250  Tpetra::Distributor source_distributor( d_comm );
251  GlobalOrdinal num_source_elements =
252  source_distributor.createFromSends( target_destinations() );
253  d_source_elements.resize( num_source_elements );
254  source_distributor.doPostsAndWaits( rendezvous_elements_view, 1,
255  d_source_elements() );
256 
257  // Send the rendezvous point coordinates to the source decomposition via
258  // inverse communication.
259  Teuchos::ArrayView<const double> rendezvous_coords_view =
260  rendezvous_coords.get1dView()();
261  d_target_coords.resize( num_source_elements*coord_dim );
262  source_distributor.doPostsAndWaits( rendezvous_coords_view, coord_dim,
263  d_target_coords() );
264 
265  // Send the rendezvous point global ordinals to the source decomposition
266  // via inverse communication.
267  Teuchos::Array<GlobalOrdinal> source_points( num_source_elements );
268  source_distributor.doPostsAndWaits( rendezvous_points_view, 1,
269  source_points() );
270 
271  // Build the data export map from the coordinate ordinals as well as
272  // populate the source element/target coordinate pairings.
273  Teuchos::ArrayView<const GlobalOrdinal> source_points_view =
274  source_points();
275  d_export_map = Tpetra::createNonContigMap<GlobalOrdinal>(
276  source_points_view, d_comm );
277  testPostcondition( d_export_map != Teuchos::null,
278  "Error creating data export map." );
279 
280  // Build the data importer.
281  d_data_export = Teuchos::rcp( new Tpetra::Export<GlobalOrdinal>(
282  d_export_map, d_import_map ) );
283  testPostcondition( d_data_export != Teuchos::null,
284  "Error creating data importer." );
285 }

291 template<class Mesh, class CoordinateField>
292 template<class SourceField, class TargetField>
293 void ConsistentEvaluation<Mesh,CoordinateField>::apply(
294  const Teuchos::RCP< FieldEvaluator<Mesh,SourceField> >& source_evaluator,
295  TargetField& target_space )
296 {
297  typedef FieldTraits<SourceField> SFT;
298  typedef FieldTraits<TargetField> TFT;
299  SourceField evaluated_field =
300  source_evaluator->evaluate( Teuchos::arcpFromArray( d_source_elements ),
301  Teuchos::arcpFromArray( d_target_coords ) );
302  testPrecondition( SFT::dim( evaluated_field ) == TFT::dim( target_space ),
303  "Source field dimension != target field dimension." );
304 
305  Teuchos::ArrayRCP<typename SFT::value_type> source_field_view;
306  if ( SFT::size( evaluated_field ) == 0 )
307  {
308  source_field_view =
309  Teuchos::ArrayRCP<typename SFT::value_type>( 0, 0.0 );
310  }
311  else
312  {
313  source_field_view =
314  FieldTools<SourceField>::nonConstView( evaluated_field );
315  }
316 
317  GlobalOrdinal source_size = SFT::size( evaluated_field ) /
318  SFT::dim( evaluated_field );
319 
320  Teuchos::RCP< Tpetra::MultiVector<typename SFT::value_type,
321  GlobalOrdinal> > source_vector =
322  createMultiVectorFromView( d_export_map,
323  source_field_view,
324  source_size,
325  SFT::dim( evaluated_field ) );
326 
327  Teuchos::ArrayRCP<typename TFT::value_type> target_field_view;
328  if ( TFT::size( target_space ) == 0 )
329  {
330  target_field_view =
331  Teuchos::ArrayRCP<typename TFT::value_type>( 0, 0.0 );
332  }
333  else
334  {
335  target_field_view =
336  FieldTools<TargetField>::nonConstView( target_space );
337  }
338 
339  GlobalOrdinal target_size = TFT::size( target_space ) /
340  TFT::dim( target_space );
341 
342  Teuchos::RCP< Tpetra::MultiVector<typename TFT::value_type,
343  GlobalOrdinal> > target_vector =
344  createMultiVectorFromView( d_import_map,
345  target_field_view,
346  target_size,
347  TFT::dim( target_space ) );
348 
349  target_vector->doExport( *source_vector, *d_data_export, Tpetra::INSERT );
350 }

357 template<class Mesh, class CoordinateField>
358 Teuchos::Array<
359  typename ConsistentEvaluation<Mesh,CoordinateField>::GlobalOrdinal>
360 ConsistentEvaluation<Mesh,CoordinateField>::computePointOrdinals(
361  const CoordinateField& coordinate_field )
362 {
363  int comm_rank = d_comm->getRank();
364  int point_dim = CFT::dim( coordinate_field );
365  GlobalOrdinal local_size =
366  std::distance( CFT::begin( coordinate_field ),
367  CFT::end( coordinate_field ) ) / point_dim;
368 
369  GlobalOrdinal global_size;
370  Teuchos::reduceAll<int,GlobalOrdinal>( *d_comm,
371  Teuchos::REDUCE_MAX,
372  1,
373  &local_size,
374  &global_size );
375 
376  Teuchos::Array<GlobalOrdinal> point_ordinals( local_size );
377  for ( GlobalOrdinal n = 0; n < local_size; ++n )
378  {
379  point_ordinals[n] = comm_rank*global_size + n;
380  }
381  return point_ordinals;
382 }
