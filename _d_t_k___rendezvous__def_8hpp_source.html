86 template<class Mesh>
87 void Rendezvous<Mesh>::build( const RCP_MeshManager& mesh_manager )
88 {
89  // Get the dimension for the mesh.
90  d_node_dim = mesh_manager->dim();
91 
92  // Extract the mesh nodes and elements that are in the bounding box. These
93  // are the pieces of the mesh that will be repartitioned.
94  getMeshInBox( mesh_manager, d_global_box );
95 
96  // Construct the rendezvous partitioning for the mesh with RCB using the
97  // nodes that are in the box.
98  d_rcb = Teuchos::rcp( new RCB<Mesh>( mesh_manager ) );
99  testPostcondition( d_rcb != Teuchos::null,
100  "Error creating RCB decomposition." );
101  d_rcb->partition();
102 
103  // Send the mesh in the box to the rendezvous decomposition and build the
104  // concrete mesh blocks.
105  MeshManager<MeshContainerType> rendezvous_mesh_manager =
106  sendMeshToRendezvous( mesh_manager );
107 
108  // Build the concrete rendezvous mesh from the mesh container.
109  d_rendezvous_mesh = createRendezvousMesh( rendezvous_mesh_manager );
110  testPostcondition( d_rendezvous_mesh != Teuchos::null,
111  "Error creating rendezvous mesh." );
112 
113  // Create a kD-tree in the rendezvous decomposition.
114  d_kdtree = Teuchos::rcp(
115  new KDTree<GlobalOrdinal>( d_rendezvous_mesh , d_node_dim ) );
116  testPostcondition( d_kdtree != Teuchos::null,
117  "Error creating rendezvous kD-tree." );
118  d_kdtree->build();
119 }

126 template<class Mesh>
127 Teuchos::Array<int> Rendezvous<Mesh>::getRendezvousProcs(
128  const Teuchos::ArrayRCP<double>& coords ) const
129 {
130  Teuchos::Array<double> point( d_node_dim );
131  GlobalOrdinal num_points = coords.size() / d_node_dim;
132  int rendezvous_proc;
133  Teuchos::Array<int> destination_procs( num_points );
134  for ( GlobalOrdinal n = 0; n < num_points; ++n )
135  {
136  for ( std::size_t d = 0; d < d_node_dim; ++d )
137  {
138  point[d] = coords[ d*num_points + n ];
139  }
140  rendezvous_proc = d_rcb->getDestinationProc( point );
141  destination_procs[n] = rendezvous_proc;
142  }
143 
144  return destination_procs;
145 }

189 template<class Mesh>
190 void Rendezvous<Mesh>::getMeshInBox( const RCP_MeshManager& mesh_manager,
191  const BoundingBox& box )
192 {
193  // Expand the box by a typical mesh element length in all directions plus
194  // some tolerance. Doing this catches a few corner cases.
195  double tol = 1.0e-4;
196  Teuchos::Tuple<double,6> box_bounds = box.getBounds();
197  double box_volume = box.volume( d_node_dim );
198  GlobalOrdinal global_num_elements = mesh_manager->globalNumElements();
199  double pow = 1 / d_node_dim;
200  double typical_length = std::pow( box_volume / global_num_elements, pow );
201  for ( std::size_t d = 0; d < d_node_dim; ++d )
202  {
203  box_bounds[d] -= typical_length + tol;
204  box_bounds[d+3] += typical_length + tol;
205  }
206  BoundingBox expanded_box( box_bounds );
207 
208  // For every mesh block, get its nodes and elements that are in the
209  // expanded box.
210  Teuchos::Array<short int> nodes_in_box, elements_in_box;
211  BlockIterator block_iterator;
212  for ( block_iterator = mesh_manager->blocksBegin();
213  block_iterator != mesh_manager->blocksEnd();
214  ++block_iterator )
215  {
216  // Setup.
217  nodes_in_box.clear();
218  elements_in_box.clear();
219  int block_id = std::distance( mesh_manager->blocksBegin(),
220  block_iterator );
221 
222  // Create a map indexed by node global ordinal containing the actual
223  // node ordinal location in the array. This will give us logarithmic
224  // time access to connectivity. I should write a more general hash
225  // table to improve this access time as I'm using this strategy for
226  // most mesh operations.
227  GlobalOrdinal num_nodes = MeshTools<Mesh>::numNodes( *block_iterator );
228  std::map<GlobalOrdinal,GlobalOrdinal> node_indices;
229  typename MT::const_node_iterator node_iterator;
230  GlobalOrdinal array_index = 0;
231  for ( node_iterator = MT::nodesBegin( *block_iterator );
232  node_iterator != MT::nodesEnd( *block_iterator );
233  ++node_iterator )
234  {
235  node_indices[ *node_iterator ] = array_index;
236  ++array_index;
237  }
238 
239  // Get all of the nodes that are in the box.
240  Teuchos::Array<double> node_coords( d_node_dim );
241  Teuchos::ArrayRCP<const double> mesh_coords =
242  MeshTools<Mesh>::coordsView( *block_iterator );
243  for ( GlobalOrdinal n = 0; n < num_nodes; ++n )
244  {
245  for ( std::size_t d = 0; d < d_node_dim; ++d )
246  {
247  node_coords[d] = mesh_coords[ d*num_nodes + n ];
248  }
249  nodes_in_box.push_back( expanded_box.pointInBox( node_coords ) );
250  }
251  assert( (GlobalOrdinal) nodes_in_box.size() == num_nodes );
252 
253  // For those nodes that are in the box, get the elements that they
254  // construct. These elements are in the box.
255  GlobalOrdinal num_elements =
256  MeshTools<Mesh>::numElements( *block_iterator );
257  std::size_t nodes_per_element = MT::nodesPerElement( *block_iterator );
258  GlobalOrdinal node_index;
259  GlobalOrdinal node_ordinal;
260  int this_element_in_box;
261  Teuchos::ArrayRCP<const GlobalOrdinal> mesh_connectivity =
262  MeshTools<Mesh>::connectivityView( *block_iterator );
263  for ( GlobalOrdinal n = 0; n < num_elements; ++n )
264  {
265  this_element_in_box = 0;
266  for ( std::size_t i = 0; i < nodes_per_element; ++i )
267  {
268  node_ordinal = mesh_connectivity[ i*num_elements + n ];
269  node_index = node_indices.find( node_ordinal )->second;
270  if ( nodes_in_box[ node_index ] )
271  {
272  this_element_in_box = 1;
273  }
274  }
275  elements_in_box.push_back( this_element_in_box );
276  }
277  assert( (GlobalOrdinal) elements_in_box.size() == num_elements );
278 
279  // Get the nodes that belong to the elements in the box, but are not
280  // necessarily in the box themselves. These will also be used in RCB.
281  for ( GlobalOrdinal n = 0; n < num_elements; ++n )
282  {
283  if ( elements_in_box[n] )
284  {
285  for ( std::size_t i = 0; i < nodes_per_element; ++i )
286  {
287  node_ordinal = mesh_connectivity[ i*num_elements + n ];
288  node_index = node_indices.find( node_ordinal )->second;
289  nodes_in_box[ node_index ] = 1;
290  }
291  }
292  }
293 
294  // Set the active node/element data in the manager for the block.
295  mesh_manager->setActiveNodes( nodes_in_box, block_id );
296  mesh_manager->setActiveElements( elements_in_box, block_id );
297  }
298 }

305 template<class Mesh>
306 MeshManager<typename Rendezvous<Mesh>::MeshContainerType>
307 Rendezvous<Mesh>::sendMeshToRendezvous(
308  const RCP_MeshManager& mesh_manager )
309 {
310  // Setup the mesh blocks.
311  Teuchos::ArrayRCP<MeshContainerType>
312  block_containers( mesh_manager->getNumBlocks() );
313  BlockIterator block_iterator;
314  for ( block_iterator = mesh_manager->blocksBegin();
315  block_iterator != mesh_manager->blocksEnd();
316  ++block_iterator )
317  {
318  int block_id = std::distance( mesh_manager->blocksBegin(),
319  block_iterator );
320 
321  // Setup the communication patterns for moving the mesh block to the
322  // rendezvous decomposition. This will also move the node and element
323  // global ordinals to the rendezvous decomposition.
324  Teuchos::Array<GlobalOrdinal> rendezvous_nodes;
325  Teuchos::Array<GlobalOrdinal> rendezvous_elements;
326  setupImportCommunication( *block_iterator,
327  mesh_manager->getActiveElements( block_id ),
328  rendezvous_nodes, rendezvous_elements );
329 
330  // Setup export node map.
331  GlobalOrdinal num_nodes = MeshTools<Mesh>::numNodes( *block_iterator );
332  Teuchos::ArrayRCP<const GlobalOrdinal> export_node_arcp =
333  MeshTools<Mesh>::nodesView( *block_iterator );
334  Teuchos::ArrayView<const GlobalOrdinal> export_node_view =
335  export_node_arcp();
336  RCP_TpetraMap export_node_map = Tpetra::createNonContigMap<GlobalOrdinal>(
337  export_node_view, d_comm );
338  testPostcondition( export_node_map != Teuchos::null,
339  "Error creating node export map." );
340 
341  // Setup import node map.
342  Teuchos::ArrayView<const GlobalOrdinal> rendezvous_nodes_view =
343  rendezvous_nodes();
344  RCP_TpetraMap import_node_map = Tpetra::createNonContigMap<GlobalOrdinal>(
345  rendezvous_nodes_view, d_comm );
346  testPostcondition( import_node_map != Teuchos::null,
347  "Error creating node import map." );
348 
349  // Setup export element map.
350  GlobalOrdinal num_elements =
351  MeshTools<Mesh>::numElements( *block_iterator );
352  Teuchos::ArrayRCP<const GlobalOrdinal> export_element_arcp =
353  MeshTools<Mesh>::elementsView( *block_iterator );
354  Teuchos::ArrayView<const GlobalOrdinal> export_element_view =
355  export_element_arcp();
356  RCP_TpetraMap export_element_map =
357  Tpetra::createNonContigMap<GlobalOrdinal>(
358  export_element_view, d_comm );
359  testPostcondition( export_element_map != Teuchos::null,
360  "Error creating element export map." );
361 
362  // Setup import element map.
363  Teuchos::ArrayView<const GlobalOrdinal> rendezvous_elements_view =
364  rendezvous_elements();
365  RCP_TpetraMap import_element_map =
366  Tpetra::createNonContigMap<GlobalOrdinal>(
367  rendezvous_elements_view, d_comm );
368  testPostcondition( import_element_map != Teuchos::null,
369  "Error creating element import map." );
370 
371  // Setup importers.
372  Tpetra::Import<GlobalOrdinal> node_importer( export_node_map,
373  import_node_map );
374  Tpetra::Import<GlobalOrdinal> element_importer( export_element_map,
375  import_element_map );
376 
377  // Move the node coordinates to the rendezvous decomposition.
378  GlobalOrdinal num_coords = d_node_dim*num_nodes;
379  Teuchos::ArrayRCP<double> export_coords_view;
380  if ( num_coords == 0 )
381  {
382  export_coords_view = Teuchos::ArrayRCP<double>( 0, 0.0 );
383  }
384  else
385  {
386  export_coords_view =
387  MeshTools<Mesh>::coordsNonConstView( *block_iterator );
388  }
389  Teuchos::RCP< Tpetra::MultiVector<double,GlobalOrdinal> > export_coords =
390  createMultiVectorFromView( export_node_map, export_coords_view,
391  num_nodes, d_node_dim );
392  Tpetra::MultiVector<double,GlobalOrdinal>
393  import_coords( import_node_map, d_node_dim );
394  import_coords.doImport( *export_coords, node_importer, Tpetra::INSERT );
395 
396  // Move the element connectivity to the rendezvous decomposition.
397  int nodes_per_element = MT::nodesPerElement( *block_iterator );
398  GlobalOrdinal num_conn = nodes_per_element * num_elements;
399  Teuchos::ArrayRCP<GlobalOrdinal> export_conn_view;
400  if ( num_conn == 0 )
401  {
402  export_conn_view = Teuchos::ArrayRCP<GlobalOrdinal>( 0, 0.0 );
403  }
404  else
405  {
406  export_conn_view =
407  MeshTools<Mesh>::connectivityNonConstView( *block_iterator );
408  }
409  Teuchos::RCP< Tpetra::MultiVector<GlobalOrdinal,GlobalOrdinal> >
410  export_conn
411  = createMultiVectorFromView( export_element_map, export_conn_view,
412  num_elements, nodes_per_element );
413  Tpetra::MultiVector<GlobalOrdinal,GlobalOrdinal> import_conn(
414  import_element_map, nodes_per_element );
415  import_conn.doImport( *export_conn, element_importer, Tpetra::INSERT );
416 
417  // Construct the mesh block container from the collected data,
418  // effectively wrapping it with mesh traits.
419  Teuchos::ArrayRCP<GlobalOrdinal>
420  rendezvous_nodes_array( rendezvous_nodes.size() );
421  std::copy( rendezvous_nodes.begin(), rendezvous_nodes.end(),
422  rendezvous_nodes_array.begin() );
423  rendezvous_nodes.clear();
424 
425  Teuchos::ArrayRCP<GlobalOrdinal>
426  rendezvous_elements_array( rendezvous_elements.size() );
427  std::copy( rendezvous_elements.begin(), rendezvous_elements.end(),
428  rendezvous_elements_array.begin() );
429  rendezvous_elements.clear();
430 
431  Teuchos::ArrayRCP<const std::size_t> permutation_list =
432  MeshTools<Mesh>::permutationView( *block_iterator );
433 
434  block_containers[ block_id ] =
435  MeshContainerType( d_node_dim,
436  rendezvous_nodes_array,
437  import_coords.get1dView(),
438  MT::elementTopology( *block_iterator ),
439  nodes_per_element,
440  rendezvous_elements_array,
441  import_conn.get1dView(),
442  permutation_list );
443  }
444 
445  // Build the rendezvous mesh manager from the rendezvous mesh blocks.
446  return MeshManager<MeshContainerType>( block_containers,
447  d_comm,
448  d_node_dim );
449 }

456 template<class Mesh>
457 void Rendezvous<Mesh>::setupImportCommunication(
458  const Mesh& mesh,
459  const Teuchos::ArrayView<short int>& elements_in_box,
460  Teuchos::Array<GlobalOrdinal>& rendezvous_nodes,
461  Teuchos::Array<GlobalOrdinal>& rendezvous_elements )
462 {
463  // Create a node index map for logarithmic time access to connectivity
464  // data.
465  typename MT::const_node_iterator export_node_iterator;
466  std::map<GlobalOrdinal,GlobalOrdinal> node_indices;
467  GlobalOrdinal array_index = 0;
468  for ( export_node_iterator = MT::nodesBegin( mesh );
469  export_node_iterator != MT::nodesEnd( mesh );
470  ++export_node_iterator )
471  {
472  node_indices[ *export_node_iterator ] = array_index;
473  ++array_index;
474  }
475 
476  // Create a element index map for logarithmic time access to connectivity
477  // data.
478  typename MT::const_element_iterator export_element_iterator;
479  std::map<GlobalOrdinal,GlobalOrdinal> element_indices;
480  array_index = 0;
481  for ( export_element_iterator = MT::elementsBegin( mesh );
482  export_element_iterator != MT::elementsEnd( mesh );
483  ++export_element_iterator )
484  {
485  element_indices[ *export_element_iterator ] = array_index;
486  ++array_index;
487  }
488 
489  // Get destination procs for all local elements in the global bounding
490  // box. The element will need to be sent to each partition that its
491  // connecting nodes exist in. We'll make a unique destination proc set for
492  // each element.
493  GlobalOrdinal num_nodes = MeshTools<Mesh>::numNodes( mesh );
494  GlobalOrdinal num_elements = MeshTools<Mesh>::numElements( mesh );
495  Teuchos::Array< std::set<int> > export_element_procs_set( num_elements );
496  std::size_t nodes_per_element = MT::nodesPerElement( mesh );
497  GlobalOrdinal node_index;
498  GlobalOrdinal node_ordinal;
499  int destination_proc;
500  Teuchos::Array<double> node_coords( d_node_dim );
501  Teuchos::ArrayRCP<const double> mesh_coords =
502  MeshTools<Mesh>::coordsView( mesh );
503  Teuchos::ArrayRCP<const GlobalOrdinal> mesh_connectivity =
504  MeshTools<Mesh>::connectivityView( mesh );
505  for ( GlobalOrdinal n = 0; n < num_elements; ++n )
506  {
507  if ( elements_in_box[n] )
508  {
509  for ( std::size_t i = 0; i < nodes_per_element; ++i )
510  {
511  node_ordinal = mesh_connectivity[ i*num_elements + n ];
512  node_index = node_indices.find( node_ordinal )->second;
513  for ( std::size_t d = 0; d < d_node_dim; ++d )
514  {
515  node_coords[d] = mesh_coords[ d*num_nodes + node_index ];
516  }
517  destination_proc = d_rcb->getDestinationProc( node_coords );
518  export_element_procs_set[n].insert( destination_proc );
519  }
520  }
521  }
522 
523  // Unroll the vector of sets into two vectors; one containing the element
524  // ordinal and the other containing the corresponding element destination.
525  Teuchos::Array<GlobalOrdinal> export_elements;
526  Teuchos::Array<int> export_element_procs;
527  typename MT::const_element_iterator element_iterator;
528  Teuchos::Array< std::set<int> >::const_iterator element_vec_iterator;
529  std::set<int>::const_iterator element_proc_set_iterator;
530  for ( element_vec_iterator = export_element_procs_set.begin(),
531  element_iterator = MT::elementsBegin( mesh );
532  element_vec_iterator != export_element_procs_set.end();
533  ++element_vec_iterator, ++element_iterator )
534  {
535  for ( element_proc_set_iterator = element_vec_iterator->begin();
536  element_proc_set_iterator != element_vec_iterator->end();
537  ++element_proc_set_iterator )
538  {
539  export_elements.push_back( *element_iterator );
540  export_element_procs.push_back( *element_proc_set_iterator );
541  }
542  }
543  export_element_procs_set.clear();
544 
545  // Now we know where the elements need to go. Move the elements to the
546  // rendezvous decomposition through an inverse communciation operation.
547  Tpetra::Distributor element_distributor( d_comm );
548  Teuchos::ArrayView<int> export_element_procs_view = export_element_procs();
549  GlobalOrdinal num_import_elements = element_distributor.createFromSends(
550  export_element_procs_view );
551  Teuchos::ArrayView<const GlobalOrdinal> export_elements_view =
552  export_elements();
553  Teuchos::Array<GlobalOrdinal> import_elements( num_import_elements );
554  Teuchos::ArrayView<GlobalOrdinal> import_elements_view = import_elements();
555  element_distributor.doPostsAndWaits( export_elements_view, 1,
556  import_elements_view );
557 
558  // Next, move these into the rendezvous element set so that we have a
559  // unique list of the elements.
560  typename Teuchos::Array<GlobalOrdinal>::const_iterator
561  import_element_iterator;
562  std::set<GlobalOrdinal> rendezvous_elements_set;
563  for ( import_element_iterator = import_elements.begin();
564  import_element_iterator != import_elements.end();
565  ++import_element_iterator )
566  {
567  rendezvous_elements_set.insert( *import_element_iterator );
568  }
569  import_elements.clear();
570 
571  // Finally put the elements in a Teuchos::Array and get rid of the set.
572  rendezvous_elements.resize( rendezvous_elements_set.size() );
573  std::copy( rendezvous_elements_set.begin(), rendezvous_elements_set.end(),
574  rendezvous_elements.begin() );
575  rendezvous_elements_set.clear();
576 
577  // Now get the destination procs for all the nodes. This will be the same
578  // destination procs as all of their parent elements. Therefore, nodes may
579  // then also have to go to multiple procs because of this and these procs
580  // may be different than their original RCB procs.
581  GlobalOrdinal element_index;
582  typename Teuchos::Array<GlobalOrdinal>::const_iterator
583  export_elements_iterator;
584  Teuchos::Array<int>::const_iterator export_element_procs_iterator;
585  Teuchos::Array< std::set<int> > export_node_procs_set( num_nodes );
586  for ( export_elements_iterator = export_elements.begin(),
587  export_element_procs_iterator = export_element_procs.begin();
588  export_elements_iterator != export_elements.end();
589  ++export_elements_iterator, ++export_element_procs_iterator )
590  {
591  element_index =
592  element_indices.find( *export_elements_iterator )->second;
593 
594  for ( std::size_t i = 0; i < nodes_per_element; ++i )
595  {
596  node_ordinal = mesh_connectivity[ i*num_elements + element_index ];
597  node_index = node_indices.find( node_ordinal )->second;
598 
599  export_node_procs_set[ node_index ].insert(
600  *export_element_procs_iterator );
601  }
602  }
603  export_elements.clear();
604  export_element_procs.clear();
605  node_indices.clear();
606  element_indices.clear();
607 
608  // Unroll the vector of sets into two vectors; one containing the node
609  // ordinal and the other containing the corresponding node destination.
610  Teuchos::Array<GlobalOrdinal> export_nodes;
611  Teuchos::Array<int> export_node_procs;
612  Teuchos::Array< std::set<int> >::const_iterator node_vec_iterator;
613  std::set<int>::const_iterator node_proc_set_iterator;
614  for ( node_vec_iterator = export_node_procs_set.begin(),
615  export_node_iterator = MT::nodesBegin( mesh );
616  node_vec_iterator != export_node_procs_set.end();
617  ++node_vec_iterator, ++export_node_iterator )
618  {
619  for ( node_proc_set_iterator = node_vec_iterator->begin();
620  node_proc_set_iterator != node_vec_iterator->end();
621  ++node_proc_set_iterator )
622  {
623  export_nodes.push_back( *export_node_iterator );
624  export_node_procs.push_back( *node_proc_set_iterator );
625  }
626  }
627  export_node_procs_set.clear();
628 
629  // Now we know where the nodes need to go. Move the nodes to the
630  // rendezvous decomposition through an inverse communciation operation.
631  Tpetra::Distributor node_distributor( d_comm );
632  Teuchos::ArrayView<int> export_node_procs_view = export_node_procs();
633  GlobalOrdinal num_import_nodes = node_distributor.createFromSends(
634  export_node_procs_view );
635  Teuchos::ArrayView<const GlobalOrdinal> export_nodes_view = export_nodes();
636  Teuchos::Array<GlobalOrdinal> import_nodes( num_import_nodes );
637  Teuchos::ArrayView<GlobalOrdinal> import_nodes_view = import_nodes();
638  node_distributor.doPostsAndWaits( export_nodes_view, 1,
639  import_nodes_view );
640  export_nodes.clear();
641  export_node_procs.clear();
642 
643  // Next move these into the rendezvous node set so that we have a unique
644  // list of the nodes.
645  typename Teuchos::Array<GlobalOrdinal>::const_iterator
646  import_node_iterator;
647  std::set<GlobalOrdinal> rendezvous_nodes_set;
648  for ( import_node_iterator = import_nodes.begin();
649  import_node_iterator != import_nodes.end();
650  ++import_node_iterator )
651  {
652  rendezvous_nodes_set.insert( *import_node_iterator );
653  }
654  import_nodes.clear();
655 
656  // Finally put the nodes in a Teuchos::Array and get rid of the set.
657  rendezvous_nodes.resize( rendezvous_nodes_set.size() );
658  std::copy( rendezvous_nodes_set.begin(), rendezvous_nodes_set.end(),
659  rendezvous_nodes.begin() );
660  rendezvous_nodes_set.clear();
661 }
