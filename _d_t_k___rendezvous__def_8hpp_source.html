00095 template<class Mesh> 
00096 void Rendezvous<Mesh>::build( const RCP_MeshManager& mesh_manager )
00097 {
00098     // Get the dimension for the mesh.
00099     d_vertex_dim = mesh_manager->dim();
00100 
00101     // Extract the mesh vertices and elements that are in the bounding
00102     // box. These are the pieces of the mesh that will be repartitioned.
00103     getMeshInBox( mesh_manager );
00104 
00105     // Construct the rendezvous partitioning for the mesh with RCB using the
00106     // vertices that are in the box.
00107     d_rcb = Teuchos::rcp( new RCB<Mesh>( mesh_manager ) );
00108     testPostcondition( d_rcb != Teuchos::null );
00109     d_rcb->partition();
00110 
00111     // Send the mesh in the box to the rendezvous decomposition and build the
00112     // concrete mesh blocks.
00113     MeshManager<MeshContainerType> rendezvous_mesh_manager =
00114         sendMeshToRendezvous( mesh_manager );
00115 
00116     // Build the concrete rendezvous mesh from the mesh container.
00117     d_rendezvous_mesh = createRendezvousMesh( rendezvous_mesh_manager );
00118     testPostcondition( d_rendezvous_mesh != Teuchos::null );
00119 
00120     // Create a kD-tree in the rendezvous decomposition.
00121     d_kdtree = Teuchos::rcp( 
00122         new KDTree<GlobalOrdinal>( d_rendezvous_mesh , d_vertex_dim ) );
00123     testPostcondition( d_kdtree != Teuchos::null );
00124     d_kdtree->build();
00125 }

00219 template<class Mesh>
00220 void Rendezvous<Mesh>::getMeshInBox( const RCP_MeshManager& mesh_manager )
00221 {
00222     // Expand the box by a typical mesh element length in all directions plus
00223     // some tolerance. Doing this catches a few corner cases.
00224     double tol = 1.0e-4;
00225     Teuchos::Tuple<double,6> box_bounds = d_global_box.getBounds();
00226     double box_volume = d_global_box.volume( d_vertex_dim );
00227     GlobalOrdinal global_num_elements = mesh_manager->globalNumElements();
00228     double pow = 1 / d_vertex_dim;
00229     double typical_length = std::pow( box_volume / global_num_elements, pow );
00230     for ( int d = 0; d < d_vertex_dim; ++d )
00231     {
00232         box_bounds[d] -= typical_length + tol;
00233         box_bounds[d+3] += typical_length + tol;
00234     }
00235     d_global_box = BoundingBox( box_bounds );
00236 
00237     // For every mesh block, get its vertices and elements that are in the
00238     // expanded box.
00239     Teuchos::Array<short int> vertices_in_box, elements_in_box;
00240     BlockIterator block_iterator;
00241     for ( block_iterator = mesh_manager->blocksBegin();
00242           block_iterator != mesh_manager->blocksEnd();
00243           ++block_iterator )
00244     {
00245         // Setup.
00246         vertices_in_box.clear();
00247         elements_in_box.clear();
00248         int block_id = std::distance( mesh_manager->blocksBegin(), 
00249                                       block_iterator );
00250 
00251         // Create a map indexed by vertex global ordinal containing the actual
00252         // vertex ordinal location in the array. This will give us logarithmic
00253         // time access to connectivity. I should write a more general hash
00254         // table to improve this access time as I'm using this strategy for
00255         // most mesh operations.
00256         GlobalOrdinal num_vertices = 
00257             MeshTools<Mesh>::numVertices( *block_iterator );
00258         std::map<GlobalOrdinal,GlobalOrdinal> vertex_indices;
00259         typename MT::const_vertex_iterator vertex_iterator;
00260         GlobalOrdinal array_index = 0;
00261         for ( vertex_iterator = MT::verticesBegin( *block_iterator );
00262               vertex_iterator != MT::verticesEnd( *block_iterator );
00263               ++vertex_iterator )
00264         {
00265             vertex_indices[ *vertex_iterator ] = array_index;
00266             ++array_index;
00267         }
00268 
00269         // Get all of the vertices that are in the box. 
00270         Teuchos::Array<double> vertex_coords( d_vertex_dim );
00271         Teuchos::ArrayRCP<const double> mesh_coords =
00272             MeshTools<Mesh>::coordsView( *block_iterator );
00273         for ( GlobalOrdinal n = 0; n < num_vertices; ++n )
00274         {
00275             for ( int d = 0; d < d_vertex_dim; ++d )
00276             {
00277                 vertex_coords[d] = mesh_coords[ d*num_vertices + n ];
00278             }
00279             vertices_in_box.push_back( 
00280                 d_global_box.pointInBox( vertex_coords ) );
00281         }
00282         assert( (GlobalOrdinal) vertices_in_box.size() == num_vertices );
00283 
00284         // For those vertices that are in the box, get the elements that they
00285         // construct. These elements are in the box.
00286         GlobalOrdinal num_elements = 
00287             MeshTools<Mesh>::numElements( *block_iterator );
00288         int vertices_per_element = MT::verticesPerElement( *block_iterator );
00289         GlobalOrdinal vertex_index;
00290         GlobalOrdinal vertex_ordinal;
00291         int this_element_in_box;
00292         Teuchos::ArrayRCP<const GlobalOrdinal> mesh_connectivity = 
00293             MeshTools<Mesh>::connectivityView( *block_iterator );
00294         for ( GlobalOrdinal n = 0; n < num_elements; ++n )
00295         {
00296             this_element_in_box = 0;
00297             for ( int i = 0; i < vertices_per_element; ++i )
00298             {
00299                 vertex_ordinal = mesh_connectivity[ i*num_elements + n ];
00300                 vertex_index = vertex_indices.find( vertex_ordinal )->second;
00301                 if ( vertices_in_box[ vertex_index ] )
00302                 {
00303                     this_element_in_box = 1;
00304                 }
00305             }
00306             elements_in_box.push_back( this_element_in_box );
00307         }
00308         assert( (GlobalOrdinal) elements_in_box.size() == num_elements );
00309 
00310         // Get the vertices that belong to the elements in the box, but are not
00311         // necessarily in the box themselves. These will also be used in RCB.
00312         for ( GlobalOrdinal n = 0; n < num_elements; ++n )
00313         {
00314             if ( elements_in_box[n] )
00315             {
00316                 for ( int i = 0; i < vertices_per_element; ++i )
00317                 {
00318                     vertex_ordinal = mesh_connectivity[ i*num_elements + n ];
00319                     vertex_index = 
00320                         vertex_indices.find( vertex_ordinal )->second;
00321                     vertices_in_box[ vertex_index ] = 1;
00322                 }
00323             }
00324         }
00325 
00326         // Set the active vertex/element data in the manager for the block.
00327         mesh_manager->setActiveVertices( vertices_in_box, block_id );
00328         mesh_manager->setActiveElements( elements_in_box, block_id );
00329     }
00330 }
