00148 {

00150                                std::upper_bound( d_x_edges.begin(),
00151                                                  d_x_edges.end(),
00152                                                  coords[0] ) );
00153 
00154     int y_idx = std::distance( d_y_edges.begin(),
00155                                std::upper_bound( d_y_edges.begin(),
00156                                                  d_y_edges.end(),
00157                                                  coords[1] ) );
00158 
00159     int z_idx = std::distance( d_z_edges.begin(),
00160                                std::upper_bound( d_z_edges.begin(),
00161                                                  d_z_edges.end(),
00162                                                  coords[2] ) );
00163 
00164     if ( x_idx == 0 || y_idx == 0 || z_idx == 0 ||
00165          x_idx > (int) d_x_edges.size() || y_idx > (int) d_y_edges.size() ||
00166          z_idx > (int) d_z_edges.size() )
00167     {
00168         throw PointNotFound( "Point outside RCB decomposition." );
00169     }
00170 
00171     return (x_idx-1) + (y_idx-1)*(d_x_edges.size()-1) + 
00172         (z_idx-1)*(d_y_edges.size()-1)*(d_x_edges.size()-1);
