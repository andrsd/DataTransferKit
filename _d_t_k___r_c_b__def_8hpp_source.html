161 template<class Mesh>
162 int RCB<Mesh>::getNumberOfObjects( void *data, int *ierr )
163 {
164  RCP_MeshManager mesh_manager = *static_cast<RCP_MeshManager*>( data );
165  int num_nodes = 0;
166  int num_blocks = mesh_manager->getNumBlocks();
167  Teuchos::ArrayView<short int>::const_iterator active_iterator;
168  for ( int i = 0; i < num_blocks; ++i )
169  {
170  for ( active_iterator = mesh_manager->getActiveNodes( i ).begin();
171  active_iterator != mesh_manager->getActiveNodes( i ).end();
172  ++active_iterator )
173  {
174  if ( *active_iterator )
175  {
176  ++num_nodes;
177  }
178  }
179  }
180 
181  *ierr = ZOLTAN_OK;
182  return num_nodes;
183 }

189 template<class Mesh>
190 void RCB<Mesh>::getObjectList(
191  void *data, int sizeGID, int sizeLID,
192  ZOLTAN_ID_PTR globalID, ZOLTAN_ID_PTR localID,
193  int wgt_dim, float *obj_wgts, int *ierr )
194 {
195  RCP_MeshManager mesh_manager = *static_cast<RCP_MeshManager*>( data );
196  *ierr = ZOLTAN_OK;
197 
198  // Note here that the local ID is being set the the node array index.
199  Teuchos::ArrayRCP<short int>::const_iterator active_iterator;
200  typename MT::const_node_iterator gid_iterator;
201  zoltan_id_type i = 0;
202  zoltan_id_type j = 0;
203  BlockIterator block_iterator;
204  for ( block_iterator = mesh_manager->blocksBegin();
205  block_iterator != mesh_manager->blocksEnd();
206  ++block_iterator )
207  {
208  int block_id = std::distance( mesh_manager->blocksBegin(),
209  block_iterator );
210 
211  for ( gid_iterator = MT::nodesBegin( *block_iterator ),
212  active_iterator =
213  mesh_manager->getActiveNodes( block_id ).begin();
214  gid_iterator != MT::nodesEnd( *block_iterator );
215  ++gid_iterator, ++active_iterator )
216  {
217  if ( *active_iterator )
218  {
219  globalID[i] = static_cast<zoltan_id_type>( *gid_iterator );
220  localID[i] = j;
221  ++i;
222  }
223  ++j;
224  }
225  }
226 }

244 template<class Mesh>
245 void RCB<Mesh>::getGeometryList(
246  void *data, int sizeGID, int sizeLID,
247  int num_obj,
248  ZOLTAN_ID_PTR globalID, ZOLTAN_ID_PTR localID,
249  int num_dim, double *geom_vec, int *ierr )
250 {
251  RCP_MeshManager mesh_manager = *static_cast<RCP_MeshManager*>( data );
252 
253  // Get the number of active nodes.
254  std::size_t num_active_nodes = 0;
255  int num_blocks = mesh_manager->getNumBlocks();
256  Teuchos::ArrayView<short int>::const_iterator active_iterator;
257  for ( int i = 0; i < num_blocks; ++i )
258  {
259  for ( active_iterator = mesh_manager->getActiveNodes( i ).begin();
260  active_iterator != mesh_manager->getActiveNodes( i ).end();
261  ++active_iterator )
262  {
263  if ( *active_iterator )
264  {
265  ++num_active_nodes;
266  }
267  }
268  }
269 
270  // Check Zoltan for consistency.
271  std::size_t node_dim = mesh_manager->dim();
272  testInvariant( sizeGID == 1 );
273  testInvariant( sizeLID == 1 );
274  testInvariant( num_dim == (int) node_dim );
275  testInvariant( num_obj == (int) num_active_nodes );
276 
277  if ( sizeGID != 1 || sizeLID != 1 ||
278  num_dim != (int) node_dim || num_obj != (int) num_active_nodes )
279  {
280  *ierr = ZOLTAN_FATAL;
281  return;
282  }
283 
284  // Zoltan needs interleaved coordinates.
285  int n = 0;
286  Teuchos::ArrayRCP<const double> mesh_coords;
287  GlobalOrdinal num_nodes;
288  BlockIterator block_iterator;
289  for ( block_iterator = mesh_manager->blocksBegin();
290  block_iterator != mesh_manager->blocksEnd();
291  ++block_iterator )
292  {
293  int block_id = std::distance( mesh_manager->blocksBegin(),
294  block_iterator );
295  Teuchos::ArrayView<short int> active_nodes =
296  mesh_manager->getActiveNodes( block_id );
297 
298  mesh_coords = MeshTools<Mesh>::coordsView( *block_iterator );
299  num_nodes = std::distance( MT::nodesBegin( *block_iterator ),
300  MT::nodesEnd( *block_iterator ) );
301  for ( GlobalOrdinal i = 0; i < num_nodes; ++i )
302  {
303  if ( active_nodes[i] )
304  {
305  for ( std::size_t d = 0; d < node_dim; ++d )
306  {
307  geom_vec[ node_dim*n + d ] =
308  mesh_coords[ d*num_nodes + i ];
309  }
310  ++n;
311  }
312  }
313  }
314 }
