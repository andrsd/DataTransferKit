00209                            DTK_TRIANGLE ||

00212                            "Element topology does not match mesh dimension" );
00213         }
00214         else if ( d_dim == 3 )

00216             testInvariant( MT::elementTopology( *block_iterator ) == 
00217                            DTK_TETRAHEDRON ||
