00148 {

00152     double global_x_max = -Teuchos::ScalarTraits<double>::rmax();
00153     double global_y_max = -Teuchos::ScalarTraits<double>::rmax();
00154     double global_z_max = -Teuchos::ScalarTraits<double>::rmax();
00155 
00156     // Get the bounding box for each mesh block.
00157     Teuchos::Tuple<double,6> box_bounds;
00158     BoundingBox block_box;

00160     for ( block_iterator = d_mesh_blocks.begin();
00161           block_iterator != d_mesh_blocks.end();
00162           ++block_iterator )
00163     {
00164         // If the mesh block is empty, do nothing.
00165         if ( MeshTools<Mesh>::numVertices( *block_iterator ) > 0 )
00166         {
00167             block_box =
00168                 MeshTools<Mesh>::globalBoundingBox( *block_iterator, d_comm );
00169 
00170             box_bounds = block_box.getBounds();
00171 
00172             if ( box_bounds[0] < global_x_min )
00173             {
00174                 global_x_min = box_bounds[0];
00175             }
00176             if ( box_bounds[1] < global_y_min )
00177             {
00178                 global_y_min = box_bounds[1];
00179             }
00180             if ( box_bounds[2] < global_z_min )
00181             {
00182                 global_z_min = box_bounds[2];
00183             }
00184             if ( box_bounds[3] > global_x_max )
00185             {
00186                 global_x_max = box_bounds[3];
00187             }
00188             if ( box_bounds[4] > global_y_max )
00189             {
00190                 global_y_max = box_bounds[4];
00191             }
00192             if ( box_bounds[5] > global_z_max )
00193             {
00194                 global_z_max = box_bounds[5];
00195             }
00196         }
00197     }

00200                         global_x_max, global_y_max, global_z_max );
