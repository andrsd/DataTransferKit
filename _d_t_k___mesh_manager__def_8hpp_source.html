125 template<class Mesh>
126 BoundingBox MeshManager<Mesh>::globalBoundingBox()
127 {
128  double global_x_min = Teuchos::ScalarTraits<double>::rmax();
129  double global_y_min = Teuchos::ScalarTraits<double>::rmax();
130  double global_z_min = Teuchos::ScalarTraits<double>::rmax();
131  double global_x_max = -Teuchos::ScalarTraits<double>::rmax();
132  double global_y_max = -Teuchos::ScalarTraits<double>::rmax();
133  double global_z_max = -Teuchos::ScalarTraits<double>::rmax();
134 
135  // Get the bounding box for each mesh block.
136  Teuchos::Tuple<double,6> box_bounds;
137  BoundingBox block_box;
138  BlockIterator block_iterator;
139  for ( block_iterator = d_mesh_blocks.begin();
140  block_iterator != d_mesh_blocks.end();
141  ++block_iterator )
142  {
143  // If the mesh block is empty, do nothing.
144  if ( MeshTools<Mesh>::numNodes( *block_iterator ) > 0 )
145  {
146  block_box =
147  MeshTools<Mesh>::globalBoundingBox( *block_iterator, d_comm );
148 
149  box_bounds = block_box.getBounds();
150 
151  if ( box_bounds[0] < global_x_min )
152  {
153  global_x_min = box_bounds[0];
154  }
155  if ( box_bounds[1] < global_y_min )
156  {
157  global_y_min = box_bounds[1];
158  }
159  if ( box_bounds[2] < global_z_min )
160  {
161  global_z_min = box_bounds[2];
162  }
163  if ( box_bounds[3] > global_x_max )
164  {
165  global_x_max = box_bounds[3];
166  }
167  if ( box_bounds[4] > global_y_max )
168  {
169  global_y_max = box_bounds[4];
170  }
171  if ( box_bounds[5] > global_z_max )
172  {
173  global_z_max = box_bounds[5];
174  }
175  }
176  }
177 
178  return BoundingBox( global_x_min, global_y_min, global_z_min,
179  global_x_max, global_y_max, global_z_max );
180 }

186 template<class Mesh>
187 void MeshManager<Mesh>::validate()
188 {
189  BlockIterator block_iterator;
190  for ( block_iterator = d_mesh_blocks.begin();
191  block_iterator != d_mesh_blocks.end();
192  ++block_iterator )
193  {
194  // Nodes.
195  testPrecondition( 0 <= d_dim && d_dim <= 3 );
196  testPrecondition( d_dim == MT::nodeDim( *block_iterator ) );
197 
198  // Coordinates.
199 
200  // Element topology.
201  if ( d_dim == 0 )
202  {
203  testPrecondition( MT::elementTopology( *block_iterator )
204  == DTK_VERTEX );
205  }
206  else if ( d_dim == 1 )
207  {
208  testPrecondition( MT::elementTopology( *block_iterator ) ==
209  DTK_LINE_SEGMENT );
210  }
211  else if ( d_dim == 2 )
212  {
213  testPrecondition( MT::elementTopology( *block_iterator ) ==
214  DTK_TRIANGLE ||
215  MT::elementTopology( *block_iterator ) ==
216  DTK_QUADRILATERAL );
217  }
218  else if ( d_dim == 3 )
219  {
220  testPrecondition( MT::elementTopology( *block_iterator ) ==
221  DTK_TETRAHEDRON ||
222  MT::elementTopology( *block_iterator ) ==
223  DTK_HEXAHEDRON ||
224  MT::elementTopology( *block_iterator ) ==
225  DTK_PYRAMID ||
226  MT::elementTopology( *block_iterator ) ==
227  DTK_WEDGE );
228  }
229 
230  // Connectivity.
231 
232  // Permutation.
233  }
234 }
